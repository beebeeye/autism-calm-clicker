      <div class="hint">长按 3 秒重置</div>

    </div>

  </main>


  <script>

    const clicker = document.getElementById("clicker");

    const numberEl = document.getElementById("number");


    let count = 0;

    let pressTimer = null;

    let longPressTriggered = false;


    function render() {

      numberEl.textContent = count;

    }


    function animateNumber() {

      numberEl.classList.remove("bump");

      void numberEl.offsetWidth;

      numberEl.classList.add("bump");


      window.setTimeout(() => {

        numberEl.classList.remove("bump");

      }, 160);

    }


    function increment() {

      count += 1;

      render();

      animateNumber();

    }


    function reset() {

      count = 0;

      render();

      animateNumber();

    }


    function startPress() {

      longPressTriggered = false;


      clearTimeout(pressTimer);

      pressTimer = window.setTimeout(() => {

        longPressTriggered = true;

        reset();

      }, 3000);

    }


    function endPress(event) {

      clearTimeout(pressTimer);


      if (longPressTriggered) {

        event.preventDefault();

        return;

      }


      increment();

    }


    clicker.addEventListener("pointerdown", startPress);

    clicker.addEventListener("pointerup", endPress);

    clicker.addEventListener("pointercancel", () => clearTimeout(pressTimer));

    clicker.addEventListener("pointerleave", () => clearTimeout(pressTimer));


    document.addEventListener("contextmenu", (event) => event.preventDefault());


    let lastTouchEnd = 0;

    document.addEventListener(

      "touchend",

      (event) => {

        const now = Date.now();

        if (now - lastTouchEnd <= 300) {

          event.preventDefault();

        }

        lastTouchEnd = now;

      },

      { passive: false }

    );

  </script>
